#include<unistd.h>
#include<stdlib.h>
#include<sys/types.h>
using namespace std;
int main()
{
    pid_t pid;
    pid=fork();
    if(pid == 0)
    {
        cout<<"child process excuting"<<" "<<getpid()<<" "<<getppid()<<endl;
            
    }
    else if(pid > 0)
    {
        cout<<"Parent process excuting"<<" "<<getpid()<<" "<<getppid()<<endl;
    }
    else
    {
        cout<<"process not excuting"<<endl;
    }
}
